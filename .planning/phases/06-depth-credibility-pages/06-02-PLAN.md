```typescript
// ---------------------------------------------------------------------------
// Phase 6: Locale-aware post fetching
// Use these helpers in locale-specific blog routes (/aktuelles/, /en/news/)
// instead of the locale-agnostic fetchPosts() above.

/**
 * Fetch posts filtered by locale, sorted by publishDate descending.
 * Returns raw CollectionEntry<'post'> objects (not normalised Post type)
 * to avoid the APP_BLOG config dependency.
 */
export const fetchPostsByLocale = async (locale: 'de' | 'en') => {
  const { getCollection } = await import('astro:content');
  const posts = await getCollection('post', (entry) => {
    return entry.data.locale === locale && entry.data.draft !== true;
  });
  return posts.sort(
    (a, b) =>
      (b.data.publishDate?.getTime() ?? 0) - (a.data.publishDate?.getTime() ?? 0)
  );
};
```

**src/pages/aktuelles/tag/[tag]/[...page].astro** — DE tag pages with pagination:
```astro
---
import type { PaginateFunction } from 'astro';
import { getCollection } from 'astro:content';
import Layout from '~/layouts/PageLayout.astro';
import Grid from '~/components/blog/Grid.astro';
import Pagination from '~/components/blog/Pagination.astro';

export async function getStaticPaths({ paginate }: { paginate: PaginateFunction }) {
  const posts = await getCollection('post', (e) => e.data.locale === 'de' && !e.data.draft);

  // Collect all unique tags
  const tagSet = new Set<string>();
  posts.forEach((p) => p.data.tags?.forEach((t) => tagSet.add(t)));

  // For each tag, return paginated paths
  const paths: ReturnType<typeof paginate> extends Promise<infer T> ? T : never[] = [];
  for (const tag of tagSet) {
    const tagged = posts
      .filter((p) => p.data.tags?.includes(tag))
      .sort((a, b) => (b.data.publishDate?.getTime() ?? 0) - (a.data.publishDate?.getTime() ?? 0));
    const pages = paginate(tagged, {
      params: { tag: encodeURIComponent(tag.toLowerCase().replace(/\s+/g, '-')) },
      pageSize: 6,
    });
    paths.push(...(await pages));
  }
  return paths;
}

const { page } = Astro.props;
const { tag } = Astro.params;

const metadata = {
  title: `Tag: ${tag} — KONVOI Aktuelles`,
  description: `Blog-Posts zum Thema ${tag} von KONVOI.`,
```
