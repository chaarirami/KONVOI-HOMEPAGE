```astro
---
import { getCollection } from 'astro:content';
import type { CollectionEntry } from 'astro:content';
import PageLayout from '~/layouts/PageLayout.astro';
import Hero from '~/components/widgets/Hero.astro';
import Content from '~/components/widgets/Content.astro';
import CallToAction from '~/components/widgets/CallToAction.astro';
import { t } from '~/i18n/translations';

export async function getStaticPaths() {
  const entries = await getCollection('industry', (entry) => entry.data.locale === 'de');
  return entries.map((entry) => ({
    params: { slug: entry.data.slug },
    props: { entry },
  }));
}

interface Props {
  entry: CollectionEntry<'industry'>;
}

const { entry } = Astro.props;
const { title, riskProfile, relatedUseCases, metadata: entryMeta } = entry.data;

const locale = 'de';

const metadata = {
  title: entryMeta?.title ?? `${title} | KONVOI`,
  description: entryMeta?.description ?? riskProfile?.slice(0, 160),
};

// Map use-case canonicalKey to DE display label and slug
const useCaseLabelsDe: Record<string, { label: string; slug: string }> = {
  'cargo-theft':             { label: 'Ladungsdiebstahl',            slug: 'ladungsdiebstahl' },
  'diesel-theft':            { label: 'Dieseldiebstahl',             slug: 'dieseldiebstahl' },
  'equipment-theft':         { label: 'Equipmentdiebstahl',          slug: 'equipmentdiebstahl' },
  'operations-transparency': { label: 'Transparenz der Operationen', slug: 'transparenz-der-operationen' },
  'trailer-damage':          { label: 'Trailerschäden',              slug: 'trailerschaeden' },
  'driver-assaults':         { label: 'Fahrerangriffe',              slug: 'fahrerangriffe' },
  'stationary-time':         { label: 'Standzeit-Optimierung',       slug: 'standzeit-optimierung' },
};
---

<PageLayout metadata={metadata}>

  {/* Hero with risk profile headline — VERT-02 */}
  <Hero
    tagline="Branche"
    title={title}
    subtitle={riskProfile?.split('\n')[0]}
  />

  {/* Risk profile body — VERT-02 */}
  {riskProfile && (
    <Content
      id="risikoprofil"
      tagline={t('industry.risk_heading', locale)}
      title={t('industry.risk_heading', locale)}
    >
      <Fragment slot="content">
        <p class="text-lg text-muted dark:text-slate-300" set:html={riskProfile.replace(/\n/g, '<br>')} />
      </Fragment>
      <Fragment slot="bg">
        <div class="absolute inset-0 bg-blue-50 dark:bg-transparent"></div>
      </Fragment>
    </Content>
  )}

  {/* Cross-links to 2-3 relevant use cases — VERT-03 */}
  {relatedUseCases && relatedUseCases.length > 0 && (
    <section class="py-12 px-4 sm:px-6">
      <div class="max-w-5xl mx-auto">
        <h2 class="text-2xl font-bold font-heading mb-6 dark:text-gray-200">
          {t('industry.usecases_heading', locale)}
        </h2>
        <div class="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
          {relatedUseCases.map((canonicalKey) => {
            const uc = useCaseLabelsDe[canonicalKey];
            if (!uc) return null;
            return (
              <a
                href={`/anwendungen/${uc.slug}`}
                class="group rounded-xl border border-slate-200 dark:border-slate-700 p-5 hover:border-primary hover:shadow-md transition-all"
              >
                <span class="block font-semibold text-slate-800 dark:text-slate-200 group-hover:text-primary transition-colors">
                  {uc.label}
                </span>
                <span class="mt-1 block text-sm text-muted dark:text-slate-400">
                  {t('cta.learn_more', locale)} →
                </span>
              </a>
            );
          })}
        </div>
      </div>
    </section>
  )}

  {/* End CTA — VERT-04 */}
  <CallToAction
    title={t('homepage.cta_title', locale)}
    subtitle={t('homepage.cta_subtitle', locale)}
    actions={[
      {
        variant: 'primary',
        text: t('cta.book_consult', locale),
        href: 'mailto:info@konvoi.eu?subject=Beratungsanfrage',
      },
    ]}
  />
</PageLayout>
```
