**Step 1 — Create `scripts/validate-crosslinks.ts`:**

The script:
1. Reads all useCase markdown files from `src/content/useCase/`
2. Reads all industry markdown files from `src/content/industry/`
3. Collects the set of valid industry slugs (from `slug` field in industry frontmatter)
4. Collects the set of valid use-case canonicalKeys (from `canonicalKey` field in useCase frontmatter)
5. For each useCase entry: validates every entry in `relatedIndustries` array is a known industry slug
6. For each industry entry: validates every entry in `relatedUseCases` array is a known use-case canonicalKey
7. Prints all errors and exits with code 1 if any found; exits 0 if all clean

```typescript
#!/usr/bin/env tsx
// scripts/validate-crosslinks.ts
// Build-time cross-link validation for useCase ↔ industry cross-references.
// Fails the build (exit 1) if any relatedIndustries or relatedUseCases value
// does not match a real content entry slug/canonicalKey.
// Per RESEARCH.md Pitfall 3 and UC-05, VERT-03 requirements.

import { readFileSync, readdirSync } from 'fs';
import { join } from 'path';

// Simple YAML frontmatter parser — extracts content between the first two '---' lines
function parseFrontmatter(content: string): Record<string, unknown> {
  const match = content.match(/^---\n([\s\S]*?)\n---/);
  if (!match) return {};
  const yaml = match[1];
  const result: Record<string, unknown> = {};

  for (const line of yaml.split('\n')) {
    // Array items: lines starting with '  - '
    const arrayItemMatch = line.match(/^\s{2}-\s+(.+)$/);
    if (arrayItemMatch) {
      // find the last key set in result and append to its array
      const keys = Object.keys(result);
      const lastKey = keys[keys.length - 1];
      if (lastKey && Array.isArray(result[lastKey])) {
        (result[lastKey] as string[]).push(arrayItemMatch[1].trim().replace(/^['"]|['"]$/g, ''));
      }
      continue;
    }
    // Key: value lines
    const kvMatch = line.match(/^(\w[\w-]*):\s*(.*)$/);
    if (kvMatch) {
      const [, key, value] = kvMatch;
      const trimmed = value.trim();
      if (trimmed === '' || trimmed === '|' || trimmed === '>') {
        // Multi-line value or array follows — initialise as array if next lines are '  - '
        result[key] = [];
      } else {
        // Strip surrounding quotes if present
        result[key] = trimmed.replace(/^['"]|['"]$/g, '');
      }
    }
  }

  return result;
}

function getMdFiles(dir: string): string[] {
  try {
    return readdirSync(dir, { recursive: true } as Parameters<typeof readdirSync>[1])
      .filter((f): f is string => typeof f === 'string' && f.endsWith('.md'));
  } catch {
    return [];
  }
}

function readFrontmatters(baseDir: string): Array<Record<string, unknown>> {
  const files = getMdFiles(baseDir);
  return files.map((file) => {
    const fullPath = join(baseDir, file);
    const content = readFileSync(fullPath, 'utf-8');
    return { _file: file, ...parseFrontmatter(content) };
  });
}

const ROOT = process.cwd();
const useCaseDir = join(ROOT, 'src/content/useCase');
const industryDir = join(ROOT, 'src/content/industry');

const useCases = readFrontmatters(useCaseDir);
const industries = readFrontmatters(industryDir);

// Build lookup sets
const industrySlugSet = new Set<string>(
  industries
    .map((e) => e['slug'])
    .filter((s): s is string => typeof s === 'string')
);

const useCaseCanonicalKeySet = new Set<string>(
  useCases
    .map((e) => e['canonicalKey'])
    .filter((s): s is string => typeof s === 'string')
);

const errors: string[] = [];

// Validate useCase.relatedIndustries → must all be known industry slugs
for (const uc of useCases) {
  const related = uc['relatedIndustries'];
  if (!Array.isArray(related) || related.length === 0) continue;
  for (const slug of related as string[]) {
    if (!industrySlugSet.has(slug)) {
      errors.push(
        `[useCase/${uc['_file']}] relatedIndustries contains unknown industry slug: "${slug}"\n` +
        `  Known slugs: ${[...industrySlugSet].join(', ')}`
      );
    }
  }
}

// Validate industry.relatedUseCases → must all be known useCase canonicalKeys
for (const ind of industries) {
  const related = ind['relatedUseCases'];
  if (!Array.isArray(related) || related.length === 0) continue;
  for (const key of related as string[]) {
    if (!useCaseCanonicalKeySet.has(key)) {
      errors.push(
        `[industry/${ind['_file']}] relatedUseCases contains unknown useCase canonicalKey: "${key}"\n` +
        `  Known keys: ${[...useCaseCanonicalKeySet].join(', ')}`
      );
    }
  }
}

if (errors.length > 0) {
  console.error('\n❌ Cross-link validation FAILED:\n');
  for (const err of errors) {
    console.error(`  • ${err}\n`);
  }
  console.error(`${errors.length} error(s) found. Fix the above cross-links before building.\n`);
  process.exit(1);
} else {
  const ucCount = useCases.length;
  const indCount = industries.length;
  console.log(`✓ Cross-link validation passed — ${ucCount} use cases × ${indCount} industries, no broken links.`);
}
```
