import { getCollection } from 'astro:content';
import type { CollectionEntry } from 'astro:content';
import PageLayout from '~/layouts/PageLayout.astro';
import Hero from '~/components/widgets/Hero.astro';
import Content from '~/components/widgets/Content.astro';
import CallToAction from '~/components/widgets/CallToAction.astro';
import SensorDataViz from '~/components/islands/SensorDataViz';
import { t } from '~/i18n/translations';

export async function getStaticPaths() {
  const entries = await getCollection('useCase', (entry) => entry.data.locale === 'de');
  return entries.map((entry) => ({
    params: { slug: entry.data.slug },
    props: { entry },
  }));
}

interface Props {
  entry: CollectionEntry<'useCase'>;
}

const { entry } = Astro.props;
const { title, problemStatement, costAnchor, konvoiApproach, canonicalKey, relatedIndustries, metadata: entryMeta } = entry.data;

const locale = 'de';

const metadata = {
  title: entryMeta?.title ?? `${title} | KONVOI`,
  description: entryMeta?.description ?? problemStatement?.slice(0, 160),
};
---

<PageLayout metadata={metadata}>

  {/* Problem framing hero — UC-02 */}
  <Hero
    tagline="Anwendungsfall"
    title={title}
    subtitle={costAnchor}
  />

  {/* Problem section — UC-02 */}
  {problemStatement && (
    <Content
      id="das-problem"
      tagline={t('usecase.problem_heading', locale)}
      title={t('usecase.problem_heading', locale)}
    >
      <Fragment slot="content">
        <p class="text-lg text-muted dark:text-slate-300" set:html={problemStatement.replace(/\n/g, '<br>')} />
      </Fragment>
      <Fragment slot="bg">
        <div class="absolute inset-0 bg-blue-50 dark:bg-transparent"></div>
      </Fragment>
    </Content>
  )}

  {/* SensorDataViz island — UC-03, UC-04 */}
  <section class="relative py-12 px-4 sm:px-6">
    <div class="max-w-5xl mx-auto">
      <h2 class="text-2xl font-bold font-heading mb-2 dark:text-gray-200">
        {t('usecase.viz_heading', locale)}
      </h2>
      <p class="text-muted mb-6 dark:text-slate-400">{t('usecase.viz_subheading', locale)}</p>
      <SensorDataViz scenarioId={canonicalKey} locale={locale} client:visible />
    </div>
  </section>

  {/* KONVOI approach — UC-02 */}
  {konvoiApproach && (
    <Content
      id="konvoi-ansatz"
      tagline={t('usecase.approach_heading', locale)}
      title={t('usecase.approach_heading', locale)}
    >
      <Fragment slot="content">
        <p class="text-lg text-muted dark:text-slate-300" set:html={konvoiApproach.replace(/\n/g, '<br>')} />
      </Fragment>
    </Content>
  )}

  {/* Cross-links to related industries — UC-05 */}
  {relatedIndustries && relatedIndustries.length > 0 && (
    <section class="py-8 px-4 sm:px-6 border-t border-slate-200 dark:border-slate-700">
      <div class="max-w-5xl mx-auto">
        <p class="text-sm text-muted dark:text-slate-400 mb-3">Relevante Branchen:</p>
        <div class="flex flex-wrap gap-3">
          {relatedIndustries.map((industrySlug) => (
            <a
              href={`/branchen/${industrySlug}`}
              class="inline-flex items-center gap-1 rounded-full border border-primary/40 px-4 py-1.5 text-sm text-primary hover:bg-primary hover:text-white transition-colors"
            >
              {industrySlug.replace(/-/g, ' ')}
            </a>
          ))}
        </div>
      </div>
    </section>
  )}

  {/* End CTA */}
  <CallToAction
    title={t('homepage.cta_title', locale)}
    subtitle={t('homepage.cta_subtitle', locale)}
    actions={[
      {
        variant: 'primary',
        text: t('cta.book_consult', locale),
        href: 'mailto:info@konvoi.eu?subject=Beratungsanfrage',
      },
    ]}
  />
</PageLayout>
