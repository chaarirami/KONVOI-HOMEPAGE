Complete file content:

```astro
# ---
import { getCollection } from 'astro:content';
import Layout from '~/layouts/PageLayout.astro';
import Hero from '~/components/widgets/Hero.astro';
import CallToAction from '~/components/widgets/CallToAction.astro';
import IncidentVideo from '~/components/widgets/IncidentVideo.astro';

export async function getStaticPaths() {
  const useCases = await getCollection('useCase', ({ data }) => data.locale === 'de');
  return useCases.map((entry) => ({
    params: { slug: entry.data.slug },
    props: { entry },
  }));
}

const { entry } = Astro.props;
const { Content } = await entry.render();
const { data } = entry;

// Industry name and href lookup for cross-links (UC-05 per D-14)
const industryNames: Record<string, { name: string; href: string }> = {
  'high-value':  { name: 'Hochwertige Güter',     href: '/branchen/hochwertige-gueter' },
  'cooling':     { name: 'Kühltransporte',         href: '/branchen/kuehlgut' },
  'intermodal':  { name: 'Intermodaler Transport', href: '/branchen/intermodal' },
  'other':       { name: 'Weitere Transporte',     href: '/branchen/sonstige' },
};

const metadata = {
  title: `${data.title} | KONVOI`,
  description: data.problemStatement ?? `KONVOI verhindert ${data.title.toLowerCase()} durch präventive Sensor-Klassifizierung.`,
};
# ---

<Layout metadata={metadata}>

  <!-- Hero: page title and tagline -->
  <Hero
    tagline="Anwendungen"
    title={data.title}
    actions={[
      { variant: 'primary', text: 'Beratung anfragen', href: '#consult' },
    ]}
  />

  <div class="max-w-3xl mx-auto px-4 sm:px-6 py-12 space-y-12">

    <!-- Problem framing (UC-02) -->
    {data.problemStatement && (
      <section>
        <h2 class="text-3xl font-bold mb-4 font-heading">Das Problem</h2>
        <p class="text-lg text-muted leading-relaxed">{data.problemStatement}</p>
      </section>
    )}

    <!-- Cost anchor callout (UC-02) -->
    {data.costAnchor && (
      <div class="border-l-4 border-primary bg-primary/5 dark:bg-primary/10 px-6 py-4 rounded-r-lg">
        <p class="text-lg font-semibold text-foreground">{data.costAnchor}</p>
      </div>
    )}

    <!-- KONVOI approach (UC-02) -->
    {data.konvoiApproach && (
      <section>
        <h2 class="text-3xl font-bold mb-4 font-heading">Der KONVOI Ansatz</h2>
        <p class="text-lg text-muted leading-relaxed">{data.konvoiApproach}</p>
      </section>
    )}

    <!-- Full markdown body (400-600 words, D-12) -->
    <div class="prose dark:prose-invert max-w-none prose-headings:font-heading prose-a:text-primary">
      <Content />
    </div>

  </div>

  <!-- Incident video placeholder (UC-03, UC-04, D-13) -->
  <IncidentVideo
    videoSrc={`/videos/incident-${data.slug}.mp4`}
    posterImage={`/images/placeholders/incident-${data.slug}.jpg`}
    caption={`Dashboard-Aufnahme: ${data.title} — echte Vorfallrekonstruktion`}
    id="incident-video"
  />

  <!-- Related industries cross-links (UC-05, D-14) -->
  {data.relatedIndustries && data.relatedIndustries.length > 0 && (
    <section class="max-w-5xl mx-auto px-4 sm:px-6 py-12">
      <h3 class="text-2xl font-bold mb-6 font-heading">Relevant für diese Branchen</h3>
      <ul class="grid grid-cols-2 md:grid-cols-4 gap-4">
        {data.relatedIndustries.map((industrySlug) => {
          const industry = industryNames[industrySlug];
          if (!industry) return null;
          return (
            <li>
              <a
                href={industry.href}
                class="block p-4 rounded-lg border border-slate-200 dark:border-slate-700 hover:border-primary hover:bg-primary/5 transition-colors text-sm font-medium text-foreground"
              >
                {industry.name}
              </a>
            </li>
          );
        })}
      </ul>
    </section>
  )}

  <!-- End-of-page consult CTA (UC-02) -->
  <CallToAction
    id="consult"
    title="Bereit zu handeln?"
    subtitle="Lassen Sie sich von unserem Team beraten, wie KONVOI Ihre Flotte gegen dieses Risiko schützt. Kostenlos und unverbindlich."
    actions={[
      { variant: 'primary', text: 'Beratung anfragen', href: 'mailto:info@konvoi.eu?subject=Beratungsanfrage' },
      { variant: 'secondary', text: 'Fallstudien ansehen', href: '/fallstudien/' },
    ]}
  />

</Layout>
```

IMPORTANT implementation notes:
- `entry.render()` returns `{ Content }` in Astro content collections v2 (Astro 6). Use `const { Content } = await entry.render();` then render as `<Content />` inside the prose div.

- `data.slug` is used to construct the videoSrc path — this matches the convention in Plan 04 content files.
- All hrefs to `/fallstudien/` are valid Phase 6 routes (confirmed built in Phase 6).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| collection→template | useCase frontmatter rendered into page — author-controlled at build time |
| template→browser | Static HTML output — no runtime data fetching or user input |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-05-01 | Injection | Content component rendering markdown body | accept | Astro's `<Content />` renders markdown through its built-in sanitized parser. No raw `set:html` on body content. Author-controlled markdown only. |
| T-04-05-02 | Tampering | relatedIndustries hrefs constructed from collection data | mitigate | Industry hrefs are resolved through a hardcoded `industryNames` map — not interpolated directly from collection data. Invalid slugs from collection produce `null` and are filtered out. No open redirect possible. |
| T-04-05-03 | Denial of Service | videoSrc path constructed from data.slug | accept | `data.slug` is a Zod-validated string from the collection (no path traversal possible in static Astro). If video file is missing, browser shows poster image fallback — no crash, no layout break. |
| T-04-05-04 | Spoofing | Cross-links to /fallstudien/ and /en/case-studies/ | mitigate | Both routes exist from Phase 6 (verified in Phase 6 build). Use exact paths from routeMap.ts — no dynamic construction. |
</threat_model>
