// src/components/widgets/LanguageSwitcher.astro
// Functional language switcher — wired in Phase 3 (I18N-04).
// Reads current route and navigates to the translated equivalent via routeMap.
// Per D-07: never auto-redirects — locale is always explicit via URL.
import { getLocalePath } from '~/i18n/routeMap';

const currentPath = Astro.url.pathname;
const currentLocale = Astro.currentLocale ?? 'de';

// Resolve target paths via routeMap; fall back to locale roots if path not in map
const dePath = getLocalePath(currentPath, 'de') ?? '/';
const enPath = getLocalePath(currentPath, 'en') ?? '/en/';
---

<div class="flex items-center gap-1 ml-2 text-sm font-medium" aria-label="Language switcher">
  <a
    href={dePath}
    class:list={[
      'px-2 py-1 rounded transition-colors',
      currentLocale === 'de'
        ? 'text-default font-semibold underline underline-offset-2'
        : 'text-muted hover:text-default',
    ]}
    aria-current={currentLocale === 'de' ? 'page' : undefined}
    title="Deutsch"
    hreflang="de"
  >
    DE
  </a>
  <span class="text-muted" aria-hidden="true">|</span>
  <a
    href={enPath}
    class:list={[
      'px-2 py-1 rounded transition-colors',

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL pathname → LanguageSwitcher | Astro.url.pathname is server-side request context, not user input in the XSS sense — it is a normalised path string set by the Astro runtime |
| routeMap return value → href attribute | getLocalePath returns only pre-defined strings from the static routeMap or null; the fallback is a hardcoded literal ('/' or '/en/') — no user-controlled value ever reaches href |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | Tampering | html lang attribute | mitigate | `Astro.currentLocale ?? 'de'` — the fallback ensures the lang attribute is always a valid BCP-47 tag ('de' or 'en'), never empty or user-controlled. Screen readers and search engines receive a correct value. |
| T-03-04-02 | Spoofing | LanguageSwitcher href construction | accept | href values come exclusively from the static routeMap (Plan 01) or hardcoded fallback literals. No user-supplied path is ever interpolated into href. Open redirect is not possible. |
| T-03-04-03 | Information Disclosure | EN homepage stub | accept | Stub page contains no sensitive content — just a heading and a link back to DE. No PII, no internal state exposed. |
</threat_model>
